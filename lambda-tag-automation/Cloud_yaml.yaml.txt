AWSTemplateFormatVersion: '2010-09-09'
Resources:
  # Função Lambda
  TaggingLambdaFunction:
    Type: AWS::Lambda::Function
    Properties:
      Handler: lambda_function.lambda_handler
      Runtime: python3.9
      Role: !GetAtt LambdaExecutionRole.Arn
      Code:
        S3Bucket: !Sub 'lambda-deployment-${AWS::Region}'
        S3Key: 'lambda_function.zip'
  
  # Permissões de execução para a Lambda
  LambdaExecutionRole:
    Type: AWS::IAM::Role
    Properties:
      AssumeRolePolicyDocument:
        Version: '2012-10-17'
        Statement:
          - Effect: Allow
            Principal:
              Service: lambda.amazonaws.com
            Action: sts:AssumeRole
      Policies:
        - PolicyName: LambdaEC2TaggingPolicy
          PolicyDocument:
            Version: '2012-10-17'
            Statement:
              - Effect: Allow
                Action:
                  - ec2:CreateTags
                Resource: "*"
  
  # Evento CloudWatch para capturar criações de instâncias EC2
  EC2InstanceCreationRule:
    Type: AWS::Events::Rule
    Properties:
      EventPattern:
        source:
          - "aws.ec2"
        detail-type:
          - "AWS API Call via CloudTrail"
        detail:
          eventName:
            - "RunInstances"
      Targets:
        - Arn: !GetAtt TaggingLambdaFunction.Arn
          Id: "EC2TaggingLambda"
  
  # Permissão para o CloudWatch invocar a Lambda
  LambdaInvokePermission:
    Type: AWS::Lambda::Permission
    Properties:
      Action: lambda:InvokeFunction
      FunctionName: !GetAtt TaggingLambdaFunction.Arn
      Principal: events.amazonaws.com
